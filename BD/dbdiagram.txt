Project nutribot_full_documented {
  database_type: "PostgreSQL"
  Note: '''
  Arquitectura integral de Nutribot.

  OBJETIVO GENERAL:
  Esta base de datos modela un sistema conversacional nutricional
  que conversa por WhatsApp, extrae datos del usuario, mantiene
  un perfil nutricional estructurado, sugiere o registra dietas,
  almacena documentos para RAG y usa una capa semántica híbrida
  para reducir llamadas a IA.

  PRINCIPIOS DE DISEÑO:
  1) Evitar texto libre cuando el dato pertenece a un catálogo.
  2) Separar "perfil del usuario" de "orden dietética vigente".
  3) Separar "datos confirmados" de "inferencias tentativas".
  4) Usar una capa semántica barata antes de llamar a IA.
  5) Mantener trazabilidad temporal cuando sea relevante.

  ESTRATEGIA DE AHORRO DE IA:
  - Primero: normalización y alias
  - Luego: matching exacto o fuzzy textual
  - Luego: búsqueda vectorial / semántica
  - Luego: reglas nutricionales
  - Solo si sigue ambiguo: IA
  '''
}

Table usuarios {
  id bigint [pk, increment, note: 'Clave primaria interna del usuario. Todas las tablas de perfil, conversación y colas se anclan a este ID.']
  whatsapp_jid varchar(50) [not null, unique, note: 'Identificador completo del usuario en WhatsApp. Ejemplo: 51999999999@s.whatsapp.net. Sirve para integrar con el proveedor de mensajería.']
  numero_whatsapp varchar(20) [not null, unique, note: 'Número telefónico limpio del usuario, normalmente sin espacios ni símbolos. Sirve para búsquedas y para el envío de mensajes.']
  creado_en timestamp [note: 'Fecha en la que el usuario fue creado por primera vez en el sistema.']
  actualizado_en timestamp [note: 'Última vez que se actualizó algún dato de esta fila del usuario.']

  Note: '''
  TABLA: usuarios

  QUÉ ES:
  Tabla maestra central de identidad del usuario dentro del sistema.

  PARA QUÉ SIRVE:
  - Unificar perfil nutricional, conversación, memoria y colas.
  - Evitar duplicar usuarios por cada mensaje recibido.
  - Tener un ancla estable para toda la operación.

  QUÉ DEBE IR AQUÍ:
  - Solo identidad técnica y mínima del usuario.
  - No debe guardar enfermedades, dietas, alergias ni historial conversacional detallado.

  QUÉ NO DEBE IR AQUÍ:
  - Datos nutricionales
  - Resúmenes clínicos
  - Estado de conversación
  - Mensajes o documentos
  '''
}

Table incoming_messages {
  id bigint [pk, increment, note: 'Clave primaria interna del mensaje recibido.']
  provider_message_id varchar(255) [not null, unique, note: 'ID único del mensaje entregado por el proveedor de WhatsApp o el gateway. Se usa para deduplicación e idempotencia.']
  usuario_id bigint [note: 'Usuario asociado al mensaje si ya pudo resolverse. Puede venir nulo en etapas muy tempranas si aún no se identificó al usuario.']
  webhook_payload jsonb [not null, note: 'Payload bruto completo del webhook recibido. Debe guardarse tal cual para auditoría, reprocesamiento o depuración.']
  status varchar(20) [not null, default: 'pending', note: 'Estado operativo del procesamiento del mensaje: pending, processing, done, failed.']
  locked_at timestamp [note: 'Momento en el que un worker tomó este mensaje para procesarlo. Sirve para control de concurrencia y reintentos.']
  error_detail text [note: 'Último error registrado al procesar el mensaje. Solo debe usarse para diagnóstico técnico.']
  retry_count int [not null, default: 0, note: 'Número de reintentos ejecutados sobre este mensaje.']
  conflict_count int [not null, default: 0, note: 'Número de conflictos de concurrencia detectados al procesar el mensaje.']
  transient_error_count int [not null, default: 0, note: 'Número de errores temporales o recuperables, por ejemplo fallos de red o tiempo de espera.']
  created_at timestamp [note: 'Momento en que el mensaje entró al inbox del sistema.']
  updated_at timestamp [note: 'Última actualización técnica de esta fila.']

  Note: '''
  TABLA: incoming_messages

  QUÉ ES:
  Inbox operacional de mensajes entrantes.

  PARA QUÉ SIRVE:
  - Recibir mensajes del proveedor de mensajería.
  - Evitar procesamiento duplicado.
  - Permitir reintentos y trazabilidad.
  - Separar recepción del procesamiento.

  QUÉ DEBE IR AQUÍ:
  - El mensaje recibido como evento técnico.
  - El payload bruto del webhook.
  - Estado de procesamiento.

  QUÉ NO DEBE IR AQUÍ:
  - Respuestas del bot
  - Resumen conversacional
  - Datos clínicos ya estructurados
  '''
}

Table conversation_state {
  usuario_id bigint [pk, not null, note: 'Clave primaria y foránea al usuario. Hay una sola fila vigente por usuario para representar su estado actual de conversación.']
  mode varchar(30) [not null, default: 'active_chat', note: 'Modo operativo actual del diálogo. Ejemplo: active_chat o collecting_usability.']
  awaiting_field_code varchar(50) [note: 'Código del campo de perfil o formulario que el bot está esperando actualmente. Ejemplo: edad, objetivo, distrito.']
  awaiting_question_code varchar(10) [note: 'Código corto de la pregunta puntual que se espera en el flujo de usabilidad.']
  last_provider_message_id varchar(255) [note: 'ID del último mensaje entrante relevante para este estado conversacional.']
  last_turn_at timestamp [note: 'Fecha y hora del último turno efectivo del usuario o del sistema.']
  last_form_prompt_at timestamp [note: 'Última vez que se pidió explícitamente una respuesta de formulario o encuesta.']
  turns_since_last_prompt int [not null, default: 0, note: 'Cantidad de turnos transcurridos desde la última pregunta de formulario. Sirve para no insistir demasiado.']
  closure_score int [note: 'Puntaje interno que estima intención de cierre o despedida. Puede servir para decidir si pedir datos o no.']
  reply_resolved_something boolean [not null, default: false, note: 'Indica si la última respuesta del bot resolvió algo útil antes de intentar capturar más datos.']
  profile_completion_pct int [not null, default: 0, note: 'Porcentaje de avance estimado del perfil nutricional del usuario.']
  usability_completion_pct int [not null, default: 0, note: 'Porcentaje de avance estimado de la evaluación de usabilidad.']
  last_openai_response_id varchar(255) [note: 'ID del último response o cadena conversacional usada con el proveedor de IA, si aplica.']
  version int [not null, default: 1, note: 'Versión interna para control de concurrencia optimista.']
  onboarding_status varchar(20) [not null, default: 'not_started', note: 'Estado general del onboarding: not_started, invited, in_progress, completed, skipped, paused.']
  onboarding_step varchar(50) [note: 'Paso concreto del onboarding en el que está el usuario.']
  onboarding_last_invited_at timestamp [note: 'Última vez que el sistema invitó al usuario a completar onboarding.']
  onboarding_next_eligible_at timestamp [note: 'Próximo momento permitido para volver a invitar al usuario.']
  onboarding_skip_count int [not null, default: 0, note: 'Número de veces que el usuario ha omitido o rechazado el onboarding.']
  onboarding_updated_at timestamp [note: 'Última vez que cambió algo relacionado al onboarding.']
  meaningful_interactions_count int [not null, default: 0, note: 'Contador de interacciones útiles para disparar encuestas, seguimiento o milestones.']
  updated_at timestamp [note: 'Última actualización global de este estado conversacional.']

  Note: '''
  TABLA: conversation_state

  QUÉ ES:
  Estado conversacional actual del usuario.

  PARA QUÉ SIRVE:
  - Saber qué está esperando el bot.
  - Controlar onboarding y encuestas.
  - Mantener continuidad entre mensajes.
  - Evitar que el sistema pregunte cosas fuera de contexto.

  QUÉ DEBE IR AQUÍ:
  - Estado actual y operativo.
  - No historial completo.
  - No contenido clínico definitivo.

  QUÉ NO DEBE IR AQUÍ:
  - Mensajes completos
  - Enfermedades confirmadas
  - Dietas registradas como dato final
  '''
}

Table profile_extractions {
  id bigint [pk, increment, note: 'Clave primaria interna de una extracción de perfil.']
  usuario_id bigint [not null, note: 'Usuario al que pertenece esta inferencia o extracción.']
  field_code varchar(50) [not null, note: 'Nombre técnico del campo detectado. Ejemplo: edad, enfermedad, restriccion, objetivo, distrito.']
  raw_value text [not null, note: 'Texto exacto detectado en la conversación antes de normalizarlo.']
  normalized_value varchar(500) [note: 'Versión normalizada del raw_value, útil para matching semántico o fuzzy.']
  confidence decimal(4,2) [not null, note: 'Nivel de confianza interno de la extracción, entre 0.00 y 1.00.']
  evidence_text text [not null, note: 'Fragmento textual o evidencia del mensaje que justifica la extracción.']
  status varchar(20) [not null, default: 'tentative', note: 'Estado de la extracción: tentative, confirmed o rejected.']
  resolved_entity_type varchar(30) [note: 'Tipo de entidad maestra a la que se resolvió la extracción, si ya existe resolución.']
  resolved_entity_code varchar(60) [note: 'Código canónico de la entidad resuelta en el maestro correspondiente.']
  resolution_strategy varchar(20) [note: 'Estrategia usada para resolver la extracción: EXACT, TRGM, VECTOR, HYBRID, LLM.']
  semantic_cache_hit boolean [not null, default: false, note: 'Marca si la resolución vino desde cache semántico en lugar de recalcularse.']
  extracted_at timestamp [note: 'Momento en que esta extracción fue generada.']

  Note: '''
  TABLA: profile_extractions

  QUÉ ES:
  Registro de inferencias tentativas extraídas desde el chat.

  PARA QUÉ SIRVE:
  - No contaminar el perfil final con datos dudosos.
  - Permitir validación posterior.
  - Trazar de dónde salió cada inferencia.

  QUÉ DEBE IR AQUÍ:
  - Detecciones intermedias.
  - Posibles campos de perfil.
  - Señales que luego podrían confirmarse.

  QUÉ NO DEBE IR AQUÍ:
  - El dato definitivo del perfil si ya fue confirmado.
  - Prescripciones dietéticas oficiales.
  '''
}

Table extraction_jobs {
  id bigint [pk, increment, note: 'Clave primaria interna del job de extracción.']
  usuario_id bigint [not null, note: 'Usuario cuyo texto será procesado por el pipeline de extracción.']
  raw_text text [not null, note: 'Texto completo o fragmento que debe ser analizado para extraer información estructurada.']
  status varchar(20) [not null, default: 'pending', note: 'Estado del job: pending, processing, done o failed.']
  retry_count int [not null, default: 0, note: 'Número de reintentos del job.']
  locked_at timestamp [note: 'Momento en que un worker tomó este job para procesarlo.']
  created_at timestamp [note: 'Fecha de creación del job.']
  updated_at timestamp [note: 'Última actualización del job.']

  Note: '''
  TABLA: extraction_jobs

  QUÉ ES:
  Cola de trabajos para extracción semántica o estructurada del perfil.

  PARA QUÉ SIRVE:
  - Desacoplar el análisis del flujo de respuesta.
  - Procesar texto fuera del camino crítico.
  - Permitir reintentos y concurrencia controlada.

  QUÉ DEBE IR AQUÍ:
  - Trabajo pendiente de extracción.
  - Texto fuente.
  - Estado técnico del job.

  QUÉ NO DEBE IR AQUÍ:
  - Resultado final confirmado.
  '''
}

Table outgoing_messages {
  id bigint [pk, increment, note: 'Clave primaria interna del mensaje saliente.']
  idempotency_key varchar(255) [not null, unique, note: 'Clave única para evitar enviar dos veces el mismo mensaje por errores o reintentos.']
  usuario_id bigint [not null, note: 'Usuario destinatario del mensaje.']
  phone varchar(20) [not null, note: 'Número final de destino usado para el envío.']
  content_type varchar(20) [not null, note: 'Tipo de contenido: text, audio, audio_tts, buttons, list.']
  content text [not null, note: 'Contenido principal del mensaje. Puede ser texto plano o representación base del contenido.']
  status varchar(20) [not null, default: 'pending', note: 'Estado del envío: pending, processing, sending, sent, failed.']
  attempt_count int [not null, default: 0, note: 'Cantidad de intentos de envío realizados.']
  locked_at timestamp [note: 'Momento en que un worker tomó este registro para despacharlo.']
  sent_at timestamp [note: 'Momento en que el sistema marcó este mensaje como enviado.']
  created_at timestamp [note: 'Fecha de creación del mensaje saliente.']
  updated_at timestamp [note: 'Última actualización de esta fila.']
  payload_json jsonb [note: 'Payload estructurado para el proveedor si el contenido requiere botones, listas u otros formatos.']
  scheduled_at timestamp [note: 'Fecha/hora programada de envío. Permite colas diferidas.']
  last_attempt_at timestamp [note: 'Fecha del último intento de despacho.']
  error_detail text [note: 'Detalle del último error de envío.']
  provider_response jsonb [note: 'Respuesta cruda del proveedor de mensajería.']
  provider_delivery_id varchar(255) [note: 'ID de entrega devuelto por el proveedor.']

  Note: '''
  TABLA: outgoing_messages

  QUÉ ES:
  Outbox de mensajes que el sistema debe enviar.

  PARA QUÉ SIRVE:
  - Controlar idempotencia.
  - Permitir reintentos de salida.
  - Separar generación de contenido del envío real.
  - Trazar fallos y entregas.

  QUÉ DEBE IR AQUÍ:
  - Mensajes listos o pendientes de salida.
  - Estado técnico de despacho.

  QUÉ NO DEBE IR AQUÍ:
  - El historial clínico del usuario.
  - La lógica de nutrición como tal.
  '''
}

  top_candidates_json jsonb [note: 'Lista opcional de candidatos considerados durante la resolución.']

  expires_at timestamp [note: 'Fecha de expiración de la entrada de cache, si aplica.']

  actualizado_en timestamp [note: 'Última actualización del cache.']

  indexes {
    (scope, field_code, query_normalizada) [unique]
  }

  Note: '''
  TABLA: semantic_resolution_cache

  QUÉ ES:

  PARA QUÉ SIRVE:
  - Evitar recalcular fuzzy, vector o IA para la misma consulta.
  - Reducir latencia.
  - Bajar costo de llamadas al modelo.

  IMPORTANTE:
  Esta tabla es clave en la estrategia de ahorro de IA.
  '''
}

  Note: '''
  TABLA: semantic_match_log

  QUÉ ES:
